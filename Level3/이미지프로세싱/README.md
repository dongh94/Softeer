# 풀이 과정
DFS는 Stack이다.
1. 초깃값 [] 가 없는게 DFS
2. while Stack 없는게 DFS
3. append 없고 거기에 재귀 도는게 DFS

# Loop
def color(x, y, c):
    old_c = image[x][y]
    image[x][y] = c

    stack = [(x, y)]
    while stack:
        x, y = stack.pop()
        if image[x-1][y] == old_c:
            image[x-1][y] = c
            stack.append((x-1, y))
        if image[x][y-1] == old_c:
            image[x][y-1] = c
            stack.append((x, y-1))
        if x < h and image[x+1][y] == old_c:
            image[x+1][y] = c
            stack.append((x+1, y))
        if y < w and image[x][y+1] == old_c:
            image[x][y+1] = c
            stack.append((x, y+1))
